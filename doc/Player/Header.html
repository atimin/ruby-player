def Header.decode(str)
  dev_addr = DevAddr.decode(str[0,16])
  type, subtype, time, seq, size = str[16,24].unpack("NNGNN")
  Header.new(dev_addr: dev_addr, type: type, subtype: subtype, time: time, seq: seq, size: size)
end
